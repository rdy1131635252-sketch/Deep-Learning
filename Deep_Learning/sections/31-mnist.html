数据集准备

MNIST 自带 60,000 训练 + 10,000 测试 28×28 灰度图。torchvision 一行下载:

import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.data import DataLoader
from torchvision import datasets, transforms

# 数据预处理:转 tensor + 标准化
transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.1307,), (0.3081,))   # MNIST 全局均值 / 标准差
])

train_set = datasets.MNIST('./data', train=True, download=True, transform=transform)
test_set  = datasets.MNIST('./data', train=False, transform=transform)

train_loader = DataLoader(train_set, batch_size=64, shuffle=True)
test_loader  = DataLoader(test_set, batch_size=1000)

定义 CNN

一个迷你 CNN,2 个卷积块 + 2 个全连接,大概 22 万参数。

class MnistCNN(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)    # 28→28
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)   # 14→14
        self.fc1   = nn.Linear(64*7*7, 128)
        self.fc2   = nn.Linear(128, 10)
        self.drop  = nn.Dropout(0.5)

    def forward(self, x):
        x = F.max_pool2d(F.relu(self.conv1(x)), 2)     # 28→14
        x = F.max_pool2d(F.relu(self.conv2(x)), 2)     # 14→7
        x = x.flatten(1)
        x = F.relu(self.fc1(x))
        x = self.drop(x)
        return self.fc2(x)                              # raw logits

训练 + 评估

device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
model = MnistCNN().to(device)
optimizer = torch.optim.AdamW(model.parameters(), lr=1e-3)
criterion = nn.CrossEntropyLoss()       # 内置 Softmax + CE

def train_one_epoch(epoch):
    model.train()
    for batch_idx, (data, target) in enumerate(train_loader):
        data, target = data.to(device), target.to(device)
        optimizer.zero_grad()                   # 清梯度!
        output = model(data)
        loss = criterion(output, target)
        loss.backward()                          # 反向传播
        optimizer.step()                         # 更新参数

        if batch_idx % 200 == 0:
            print(f'epoch {epoch} [{batch_idx*64}/{len(train_set)}] loss={loss.item():.4f}')

def evaluate():
    model.eval()                                 # 关闭 Dropout, BN 用滑动平均
    correct = 0
    with torch.no_grad():                        # 推理时不存梯度,省显存
        for data, target in test_loader:
            data, target = data.to(device), target.to(device)
            output = model(data)
            pred = output.argmax(dim=1)
            correct += pred.eq(target).sum().item()
    acc = 100. * correct / len(test_set)
    print(f'测试准确率: {correct}/{len(test_set)} ({acc:.2f}%)')
    return acc

for epoch in range(1, 6):
    train_one_epoch(epoch)
    evaluate()

训练流程的固定模板

每一个 PyTorch 训练循环都是这个套路。背下来:

  1. model.train()
  2. optimizer.zero_grad() ← 别忘!不然梯度累加
  3. output = model(data) ← 前向
  4. loss = criterion(output, target)
  5. loss.backward() ← 反向
  6. optimizer.step() ← 更新

评估时:

  1. model.eval()
  2. with torch.no_grad():
  3. 计算指标

常见的坑

小心这些细节
  • 忘了 zero_grad() → 梯度累加,训练发散
  • 评估时忘了 model.eval() → Dropout 仍然激活,准确率掉
  • 评估时忘了 torch.no_grad() → 显存炸,速度慢
  • 用 nn.CrossEntropyLoss 时网络末尾又加了 softmax → 双重 softmax,梯度奇怪
  • 数据没有用.to(device),模型在 GPU 数据在 CPU → 报错

试一试改进

  • 加数据增强:RandomRotation(10), RandomAffine(0, translate=(0.1, 0.1))
  • 换更大模型:多加一个卷积块,看准确率能不能上 99.5%
  • 改用 Mixup:训练更稳健
  • 学习率调度:加 cosine 调度
